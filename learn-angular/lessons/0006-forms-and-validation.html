Forms & Validation

You can navigate between views. Now let's learn how to collect and validate user input with Angular's powerful forms system.

Key Insight: Angular provides two approaches to forms: Template-driven (simpler, less code) and Reactive (more powerful, testable). Understanding when to use each is crucial.

Template-driven Forms

Good for simple forms with minimal logic:

// Import FormsModule
import { FormsModule } from '@angular/forms';

@Component({
  imports: [FormsModule],
  template: `
    <form #loginForm="ngForm" (ngSubmit)="onSubmit(loginForm)">
      <div>
        <label for="email">Email</label>
        <input 
          type="email" 
          id="email"
          name="email"
          ngModel
          required
          email
          #email="ngModel"
        >
        @if (email.invalid && email.touched) {
          <div class="error">
            @if (email.errors?.['required']) {
              Email is required
            }
            @if (email.errors?.['email']) {
              Please enter a valid email
            }
          </div>
        }
      </div>
      
      <div>
        <label for="password">Password</label>
        <input 
          type="password" 
          id="password"
          name="password"
          ngModel
          required
          minlength="8"
          #password="ngModel"
        >
        @if (password.invalid && password.touched) {
          <div class="error">Password must be at least 8 characters</div>
        }
      </div>
      
      <button type="submit" [disabled]="loginForm.invalid">
        Login
      </button>
    </form>
  `
})
export class LoginComponent {
  onSubmit(form: NgForm) {
    if (form.valid) {
      console.log('Form submitted:', form.value);
    }
  }
}
ngModel caveat: Each form control needs a unique name attribute. Angular uses it to register the control.

Reactive Forms

Better for complex forms, dynamic controls, and testability:

// Import ReactiveFormsModule
import { ReactiveFormsModule, FormBuilder, Validators } from '@angular/forms';

@Component({
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="profileForm" (ngSubmit)="onSubmit()">
      <div formGroupName="personalInfo">
        <div>
          <label for="firstName">First Name</label>
          <input id="firstName" formControlName="firstName">
          @if (profileForm.get('personalInfo.firstName')?.invalid && 
               profileForm.get('personalInfo.firstName')?.touched) {
            <div class="error">First name is required</div>
          }
        </div>
        
        <div>
          <label for="lastName">Last Name</label>
          <input id="lastName" formControlName="lastName">
        </div>
      </div>
      
      <div formGroupName="address">
        <div>
          <label for="street">Street</label>
          <input id="street" formControlName="street">
        </div>
        <div>
          <label for="city">City</label>
          <input id="city" formControlName="city">
        </div>
      </div>
      
      <button type="submit" [disabled]="profileForm.invalid">
        Update Profile
      </button>
    </form>
  `
})
export class ProfileComponent implements OnInit {
  private fb = inject(FormBuilder);
  
  profileForm = this.fb.group({
    personalInfo: this.fb.group({
      firstName: ['', Validators.required],
      lastName: ['', Validators.required]
    }),
    address: this.fb.group({
      street: [''],
      city: ['']
    })
  });
  
  ngOnInit() {
    // Patch entire form
    this.profileForm.patchValue({
      personalInfo: {
        firstName: 'John',
        lastName: 'Doe'
      }
    });
  }
  
  onSubmit() {
    if (this.profileForm.valid) {
      console.log('Form value:', this.profileForm.value);
    }
  }
}
FormBuilder: The FormBuilder service reduces boilerplate. It's syntactic sugar for creating form groups and controls.

Validation

Built-in Validators

import { Validators } from '@angular/forms';

const form = this.fb.group({
  // Required field
  name: ['', Validators.required],
  
  // Email validation
  email: ['', [Validators.required, Validators.email]],
  
  // String length
  password: ['', [
    Validators.required,
    Validators.minLength(8),
    Validators.maxLength(50)
  ]],
  
  // Pattern (regex)
  phone: ['', Validators.pattern(/^\d{10}$/)],
  
  // Numeric
  age: ['', [Validators.required, Validators.min(18), Validators.max(120)]]
});

Custom Validators

// Custom validator function
export function passwordStrengthValidator(): ValidatorFn {
  return (control: AbstractControl): ValidationErrors | null => {
    const value = control.value;
    if (!value) return null;
    
    const hasUpperCase = /[A-Z]/.test(value);
    const hasLowerCase = /[a-z]/.test(value);
    const hasNumeric = /[0-9]/.test(value);
    
    const passwordValid = hasUpperCase && hasLowerCase && hasNumeric;
    
    return passwordValid ? null : {
      passwordStrength: {
        hasUpperCase,
        hasLowerCase,
        hasNumeric
      }
    };
  };
}

// Async validator (e.g., check username availability)
export function uniqueUsernameValidator(
  userService: UserService
): AsyncValidatorFn {
  return (control: AbstractControl) => {
    return timer(500).pipe(
      switchMap(() => userService.checkUsername(control.value)),
      map(isAvailable => isAvailable ? null : { usernameTaken: true }),
      take(1)
    );
  };
}

// Usage
const form = this.fb.group({
  username: ['', 
    [Validators.required], 
    [uniqueUsernameValidator(this.userService)]
  ],
  password: ['', [Validators.required, passwordStrengthValidator()]]
});
Practice Exercise: Create a registration form with:
1. Email (required, valid format)
2. Password (required, min 8 chars, custom strength validator)
3. Confirm password (must match password)
4. Username (required, async validator for uniqueness)

Dynamic Forms

Add or remove controls dynamically:

@Component({
  template: `
    <form [formGroup]="form">
      <div formArrayName="aliases">
        @for (alias of aliases.controls; track $index; let i = $index) {
          <div>
            <input [formControlName]="i" placeholder="Alias">
            <button type="button" (click)="removeAlias(i)">Remove</button>
          </div>
        }
      </div>
      
      <button type="button" (click)="addAlias()">Add Alias</button>
    </form>
  `
})
export class DynamicFormComponent {
  private fb = inject(FormBuilder);
  
  form = this.fb.group({
    aliases: this.fb.array([])
  });
  
  get aliases() {
    return this.form.get('aliases') as FormArray;
  }
  
  addAlias() {
    this.aliases.push(this.fb.control(''));
  }
  
  removeAlias(index: number) {
    this.aliases.removeAt(index);
  }
}

Form State

// Check form state
console.log(this.form.valid);        // true if all controls valid
console.log(this.form.invalid);      // true if any control invalid
console.log(this.form.pristine);     // true if user hasn't interacted
console.log(this.form.dirty);        // true if user has interacted
console.log(this.form.touched);      // true if any control touched
console.log(this.form.untouched);    // true if no control touched

// Get specific control state
const emailControl = this.form.get('email');
console.log(emailControl?.valid);
console.log(emailControl?.errors);    // { required: true, email: true }

// Disable/enable controls
emailControl?.disable();
emailControl?.enable();

// Reset form
this.form.reset();
this.form.reset({ email: '', password: '' }); // With initial values

Template-driven vs Reactive

Aspect Template-driven Reactive
Setup Less code, simpler More code, explicit
Validation Template-based Code-based
Testing Harder to test Easier to test
Dynamic controls Limited Full support
Complexity Simple forms Complex forms
