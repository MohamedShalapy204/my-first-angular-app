Angular Universal & Server-Side Rendering

You can deploy applications. Now let's learn how to use Angular Universal for server-side rendering (SSR) to improve SEO, performance, and user experience.

Key Insight: SSR renders your Angular app on the server before sending it to the browser. This improves initial load time, SEO, and social media sharing.

Why SSR?

Benefit Without SSR With SSR
SEO JavaScript required for content Content visible to crawlers
Initial Load Blank screen until JS loads Content visible immediately
Social Sharing No metadata for link previews Full metadata available
Performance Client-side rendering Faster First Contentful Paint

Setting Up Angular Universal

# Add SSR to existing project
ng add @angular/ssr

# Or create new project with SSR
ng new my-app --ssr
// app.config.ts
import { provideServerRendering } from '@angular/platform-server';

export const appConfig: ApplicationConfig = {
  providers: [
    provideServerRendering(),
    // ... other providers
  ]
};

Hydration

// app.config.ts
import { provideClientHydration } from '@angular/platform-browser';

export const appConfig: ApplicationConfig = {
  providers: [
    provideServerRendering(),
    provideClientHydration(),
    // ... other providers
  ]
};
Hydration: Angular reuses the server-rendered DOM and attaches event listeners. This avoids re-rendering the entire app on the client.

Transfer State

// Avoid duplicate HTTP requests
import { TransferState, makeStateKey } from '@angular/platform-browser';

const USERS_KEY = makeStateKey<User[]>('users');

@Component({...})
export class UserListComponent implements OnInit {
  private transferState = inject(TransferState);
  private http = inject(HttpClient);
  
  users = signal<User[]>([]);
  
  ngOnInit() {
    // Check if data exists in transfer state
    const cachedUsers = this.transferState.get<User[]>(USERS_KEY, null);
    
    if (cachedUsers) {
      // Use cached data from server
      this.users.set(cachedUsers);
    } else {
      // Fetch fresh data
      this.http.get<User[]>('/api/users').subscribe(users => {
        this.users.set(users);
        
        // Store in transfer state for client
        this.transferState.set(USERS_KEY, users);
      });
    }
  }
}

Server-Side Data Loading

// Using Resolve for SSR
import { ResolveFn } from '@angular/router';
import { TransferState, makeStateKey } from '@angular/platform-browser';

export const usersResolver: ResolveFn<User[]> = (route, injector) => {
  const transferState = injector.get(TransferState);
  const http = injector.get(HttpClient);
  
  const key = makeStateKey<User[]>('users');
  const cached = transferState.get(key, null);
  
  if (cached) {
    return of(cached);
  }
  
  return http.get<User[]>('/api/users').pipe(
    tap(users => transferState.set(key, users))
  );
};

// Route definition
{
  path: 'users',
  component: UserListComponent,
  resolve: { users: usersResolver }
}

Prerendering

SSR Considerations

// ❌ These won't work on server
window.localStorage  // Server has no window
document.getElementById('app')  // Server has no document
navigator.geolocation  // Server has no navigator

// ✅ Use PLATFORM_ID to check environment
import { PLATFORM_ID } from '@angular/core';
import { isPlatformBrowser } from '@angular/common';

@Component({...})
export class SafeComponent {
  private platformId = inject(PLATFORM_ID);
  
  ngOnInit() {
    if (isPlatformBrowser(this.platformId)) {
      // Browser-only code
      localStorage.setItem('key', 'value');
    }
  }
}

// ✅ Use isPlatformBrowser in templates
@If (isBrowser) {
  <div>Browser-only content</div>
}
Server limitations: The server doesn't have access to browser APIs. Always check isPlatformBrowser before using browser-specific code.

Deployment

# Build for SSR
ng build

# The output includes:
# - dist/my-app/browser/ (client files)
# - dist/my-app/server/ (server files)

# Run the server
node dist/my-app/server/server.js

# Or use Angular CLI
ng serve --ssr
