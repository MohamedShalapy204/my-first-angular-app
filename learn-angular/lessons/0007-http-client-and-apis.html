HTTP Client & APIs

You can collect user input with forms. Now let's learn how to send that data to servers and fetch data from APIs with Angular's HTTP client.

Key Insight: Angular's HttpClient provides a modern, observable-based API for HTTP requests. It handles parsing, error handling, and even interceptors for cross-cutting concerns.

Setting Up HttpClient

// app.config.ts
import { provideHttpClient, withInterceptors } from '@angular/common/http';
import { authInterceptor } from './interceptors/auth.interceptor';

export const appConfig: ApplicationConfig = {
  providers: [
    provideHttpClient(
      withInterceptors([authInterceptor])
    ),
    // ... other providers
  ]
};

Basic HTTP Requests

@Injectable({ providedIn: 'root' })
export class UserService {
  private http = inject(HttpClient);
  private apiUrl = 'https://api.example.com/users';
  
  // GET request
  getUsers(): Observable<User[]> {
    return this.http.get<User[]>(this.apiUrl);
  }
  
  // GET with params
  getUser(id: number): Observable<User> {
    return this.http.get<User>(`${this.apiUrl}/${id}`);
  }
  
  // POST request
  createUser(user: User): Observable<User> {
    return this.http.post<User>(this.apiUrl, user);
  }
  
  // PUT request (full update)
  updateUser(id: number, user: User): Observable<User> {
    return this.http.put<User>(`${this.apiUrl}/${id}`, user);
  }
  
  // PATCH request (partial update)
  patchUser(id: number, changes: Partial<User>): Observable<User> {
    return this.http.patch<User>(`${this.apiUrl}/${id}`, changes);
  }
  
  // DELETE request
  deleteUser(id: number): Observable<void> {
    return this.http.delete<void>(`${this.apiUrl}/${id}`);
  }
}
Type Safety: Always specify the response type with generics (get<User[]>). This gives you compile-time type checking.

Request Options

// Headers
const headers = new HttpHeaders({
  'Content-Type': 'application/json',
  'Authorization': `Bearer ${token}`
});

this.http.get<User[]>(this.apiUrl, { headers });

// Query parameters
const params = new HttpParams()
  .set('page', '1')
  .set('limit', '10')
  .set('search', 'angular');

this.http.get<User[]>(this.apiUrl, { params });

// Combined options
this.http.get<User[]>(this.apiUrl, {
  headers,
  params,
  observe: 'response',  // Get full response
  responseType: 'json'  // Default
});

// With timeout
this.http.get<User[]>(this.apiUrl).pipe(
  timeout(5000)  // 5 second timeout
);

Error Handling

getUsers(): Observable<User[]> {
  return this.http.get<User[]>(this.apiUrl).pipe(
    catchError(this.handleError)
  );
}

private handleError(error: HttpErrorResponse): Observable<never> {
  let errorMessage = 'Unknown error occurred';
  
  if (error.error instanceof ErrorEvent) {
    // Client-side error
    errorMessage = `Client error: ${error.error.message}`;
  } else {
    // Server-side error
    errorMessage = `Server error: ${error.status} - ${error.message}`;
  }
  
  console.error(errorMessage);
  return throwError(() => new Error(errorMessage));
}

// In component
this.userService.getUsers().subscribe({
  next: (users) => this.users.set(users),
  error: (err) => this.error.set(err.message),
  complete: () => this.loading.set(false)
});
Always handle errors: HTTP requests can fail for many reasons (network, server, timeout). Always provide error handling in your subscriptions.

HTTP Interceptors

Interceptors modify requests/responses globally:

// Functional interceptor (Angular 15+)
import { HttpInterceptorFn } from '@angular/common/http';

export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const token = localStorage.getItem('auth_token');
  
  if (token) {
    const cloned = req.clone({
      setHeaders: {
        Authorization: `Bearer ${token}`
      }
    });
    return next(cloned);
  }
  
  return next(req);
};

// Class-based interceptor (older style)
@Injectable()
export class LoggingInterceptor implements HttpInterceptor {
  intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
    console.log(`HTTP ${req.method} ${req.url}`);
    
    return next.handle(req).pipe(
      tap(event => {
        if (event instanceof HttpResponse) {
          console.log(`Response status: ${event.status}`);
        }
      })
    );
  }
}

Common Interceptor Patterns

  • Authentication: Add auth tokens to requests
  • Logging: Log requests and responses
  • Error handling: Global error processing
  • Retry logic: Automatically retry failed requests
  • Caching: Cache GET responses

Working with Signals

Convert Observables to Signals for reactive UI:

@Component({
  template: `
    @if (loading()) {
      <p>Loading...</p>
    } @else if (error()) {
      <p class="error">{{ error() }}</p>
    } @else {
      <ul>
        @for (user of users(); track user.id) {
          <li>{{ user.name }}</li>
        }
      </ul>
    }
  `
})
export class UserListComponent {
  private userService = inject(UserService);
  
  // Using toSignal (Angular 16+)
  users = toSignal(this.userService.getUsers(), { initialValue: [] });
  loading = signal(true);
  error = signal<string | null>(null);
  
  // Or manual approach
  constructor() {
    effect(() => {
      this.userService.getUsers().subscribe({
        next: (users) => {
          this.users.set(users);
          this.loading.set(false);
        },
        error: (err) => {
          this.error.set(err.message);
          this.loading.set(false);
        }
      });
    });
  }
}

Real-World Patterns

1. Service with Caching

@Injectable({ providedIn: 'root' })
export class ProductService {
  private http = inject(HttpClient);
  private cache = new Map<string, any>();
  
  getProducts(): Observable<Product[]> {
    const cacheKey = 'products';
    
    if (this.cache.has(cacheKey)) {
      return of(this.cache.get(cacheKey));
    }
    
    return this.http.get<Product[]>('/api/products').pipe(
      tap(products => this.cache.set(cacheKey, products)),
      shareReplay(1)
    );
  }
  
  invalidateCache() {
    this.cache.clear();
  }
}

2. Upload with Progress

uploadFile(file: File): Observable<HttpEvent<any>> {
  const formData = new FormData();
  formData.append('file', file);
  
  return this.http.post('/api/upload', formData, {
    reportProgress: true,
    observe: 'events'
  });
}

// In component
this productService.uploadFile(file).subscribe(event => {
  if (event.type === HttpEventType.UploadProgress) {
    const percent = Math.round(100 * event.loaded / event.total!);
    this.uploadProgress.set(percent);
  } else if (event.type === HttpEventType.Response) {
    this.uploadComplete.set(true);
  }
});

3. Cancel Previous Requests

private searchSubject = new Subject<string>();
  
  searchResults = this.searchSubject.pipe(
    debounceTime(300),
    distinctUntilChanged(),
    switchMap(query => this.http.get<Product[]>(`/api/search?q=${query}`))
  );
  
  onSearch(query: string) {
    this.searchSubject.next(query);
  }

Testing HTTP

describe('UserService', () => {
  let service: UserService;
  let httpMock: HttpTestingController;
  
  beforeEach(() => {
    TestBed.configureTestingModule({
      providers: [
        UserService,
        provideHttpClient(),
        provideHttpClientTesting()
      ]
    });
    
    service = TestBed.inject(UserService);
    httpMock = TestBed.inject(HttpTestingController);
  });
  
  it('should fetch users', () => {
    const mockUsers = [{ id: 1, name: 'John' }];
    
    service.getUsers().subscribe(users => {
      expect(users.length).toBe(1);
      expect(users[0].name).toBe('John');
    });
    
    const req = httpMock.expectOne('/api/users');
    expect(req.request.method).toBe('GET');
    req.flush(mockUsers);
  });
  
  afterEach(() => {
    httpMock.verify(); // Ensure no unmatched requests
  });
});
