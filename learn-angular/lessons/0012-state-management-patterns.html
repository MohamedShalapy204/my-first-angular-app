State Management Patterns

You can build performant applications. Now let's learn how to manage complex application state in a predictable, maintainable way.

Key Insight: State management isn't just about storing data—it's about creating a single source of truth, enabling time-travel debugging, and making state changes predictable and testable.

State Management Spectrum

Approach Complexity Use Case
Component State Low Simple, isolated UI state
Service with Signals Medium Shared state across components
NgRx/NGXS High Complex apps, strict patterns
SignalStore Medium Modern Angular, signal-based

Service with Signals (Recommended for Most Apps)

@Injectable({ providedIn: 'root' })
export class TodoStateService {
  // Private state
  private todosSignal = signal<Todo[]>([]);
  private loadingSignal = signal(false);
  private errorSignal = signal<string | null>(null);
  
  // Public read-only signals
  readonly todos = this.todosSignal.asReadonly();
  readonly loading = this.loadingSignal.asReadonly();
  readonly error = this.errorSignal.asReadonly();
  
  // Computed state
  readonly completedCount = computed(() => 
    this.todosSignal().filter(t => t.completed).length
  );
  
  readonly stats = computed(() => ({
    total: this.todosSignal().length,
    completed: this.completedCount(),
    pending: this.todosSignal().length - this.completedCount()
  }));
  
  // Actions
  async loadTodos() {
    this.loadingSignal.set(true);
    this.errorSignal.set(null);
    
    try {
      const todos = await firstValueFrom(this.http.get<Todo[]>('/api/todos'));
      this.todosSignal.set(todos);
    } catch (error) {
      this.errorSignal.set('Failed to load todos');
    } finally {
      this.loadingSignal.set(false);
    }
  }
  
  addTodo(title: string) {
    const newTodo: Todo = {
      id: Date.now(),
      title,
      completed: false
    };
    
    this.todosSignal.update(todos => [...todos, newTodo]);
  }
  
  toggleTodo(id: number) {
    this.todosSignal.update(todos => 
      todos.map(todo => 
        todo.id === id 
          ? { ...todo, completed: !todo.completed }
          : todo
      )
    );
  }
  
  removeTodo(id: number) {
    this.todosSignal.update(todos => 
      todos.filter(todo => todo.id !== id)
    );
  }
}

Component Usage

@Component({
  template: `
    @if (todoState.loading()) {
      <p>Loading...</p>
    } @else if (todoState.error()) {
      <p class="error">{{ todoState.error() }}</p>
    } @else {
      <div class="stats">
        <p>Total: {{ todoState.stats().total }}</p>
        <p>Completed: {{ todoState.stats().completed }}</p>
      </div>
      
      @for (todo of todoState.todos(); track todo.id) {
        <app-todo-item 
          [todo]="todo" 
          (toggle)="todoState.toggleTodo($event)"
          (remove)="todoState.removeTodo($event)"
        />
      }
    }
  `
})
export class TodoListComponent {
  todoState = inject(TodoStateService);
  
  constructor() {
    this.todoState.loadTodos();
  }
}
Encapsulation: Keep state private with signals, expose only read-only versions. Components can read but not directly modify state.

NgRx Pattern (Complex Apps)

// State interface
export interface TodoState {
  todos: Todo[];
  loading: boolean;
  error: string | null;
}

// Initial state
const initialState: TodoState = {
  todos: [],
  loading: false,
  error: null
};

// Actions
export const loadTodos = createAction('[Todo] Load Todos');
export const loadTodosSuccess = createAction(
  '[Todo] Load Todos Success',
  props<{ todos: Todo[] }>()
);
export const addTodo = createAction(
  '[Todo] Add Todo',
  props<{ title: string }>()
);

// Reducer
export const todoReducer = createReducer(
  initialState,
  on(loadTodos, state => ({ ...state, loading: true })),
  on(loadTodosSuccess, (state, { todos }) => ({
    ...state,
    todos,
    loading: false
  })),
  on(addTodo, (state, { title }) => ({
    ...state,
    todos: [...state.todos, { id: Date.now(), title, completed: false }]
  }))
);

// Selectors
export const selectTodos = createSelector(
  (state: { todos: TodoState }) => state.todos.todos
);

export const selectCompletedCount = createSelector(
  selectTodos,
  todos => todos.filter(t => t.completed).length
);

SignalStore (Modern Angular)

import { signalStore, withState, withMethods, patchState } from '@ngrx/signals';

export const TodoStore = signalStore(
  { providedIn: 'root' },
  
  withState({
    todos: [] as Todo[],
    loading: false,
    error: null as string | null
  }),
  
  withMethods((store) => ({
    async loadTodos() {
      patchState(store, { loading: true, error: null });
      
      try {
        const todos = await firstValueFrom(
          inject(HttpClient).get<Todo[]>('/api/todos')
        );
        patchState(store, { todos });
      } catch (error) {
        patchState(store, { error: 'Failed to load' });
      } finally {
        patchState(store, { loading: false });
      }
    },
    
    addTodo(title: string) {
      patchState(store, state => ({
        todos: [...state.todos, { id: Date.now(), title, completed: false }]
      }));
    },
    
    toggleTodo(id: number) {
      patchState(store, state => ({
        todos: state.todos.map(todo =>
          todo.id === id ? { ...todo, completed: !todo.completed } : todo
        )
      }));
    }
  }))
);

// Usage in component
@Component({...})
export class TodoComponent {
  private store = inject(TodoStore);
  
  todos = this.store.todos;
  loading = this.store.loading;
  
  constructor() {
    this.store.loadTodos();
  }
}

When to Use Each Pattern

  • Component state: UI-only state (modals, form inputs, hover states)
  • Service with signals: Most apps, shared state, moderate complexity
  • NgRx: Large teams, strict patterns, time-travel debugging needed
  • SignalStore: Modern Angular, signal-based, less boilerplate than NgRx
Don't over-engineer: Start with service + signals. Only add NgRx when you have complex async flows, multiple reducers, or need DevTools integration.

Testing State Management

describe('TodoStateService', () => {
  let service: TodoStateService;
  
  beforeEach(() => {
    TestBed.configureTestingModule({
      providers: [TodoStateService]
    });
    service = TestBed.inject(TodoStateService);
  });
  
  it('should add todo', () => {
    service.addTodo('Test Todo');
    
    expect(service.todos().length).toBe(1);
    expect(service.todos()[0].title).toBe('Test Todo');
  });
  
  it('should toggle todo', () => {
    service.addTodo('Test Todo');
    const id = service.todos()[0].id;
    
    service.toggleTodo(id);
    
    expect(service.todos()[0].completed).toBe(true);
  });
  
  it('should compute stats', () => {
    service.addTodo('Todo 1');
    service.addTodo('Todo 2');
    service.toggleTodo(service.todos()[0].id);
    
    expect(service.stats()).toEqual({
      total: 2,
      completed: 1,
      pending: 1
    });
  });
});
