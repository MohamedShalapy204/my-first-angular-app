Angular Animations

You can build powerful components. Now let's learn how to add smooth, performant animations that enhance user experience without compromising performance.

Key Insight: Angular's animation system is built on CSS Web Animations API. It provides a declarative, framework-integrated way to create complex animations that work across browsers.

Setting Up Animations

// app.config.ts
import { provideAnimations } from '@angular/platform-browser/animations';

export const appConfig: ApplicationConfig = {
  providers: [
    provideAnimations(),
    // ... other providers
  ]
};

Basic Animation

import { trigger, state, style, animate, transition } from '@angular/animations';

@Component({
  selector: 'app-simple-animation',
  template: `
    <div [@openClose]="isOpen ? 'open' : 'closed'" class="box">
      {{ isOpen ? 'Open' : 'Closed' }}
    </div>
    
    <button (click)="isOpen = !isOpen">Toggle</button>
  `,
  animations: [
    trigger('openClose', [
      state('open', style({
        height: '200px',
        opacity: 1,
        backgroundColor: 'green'
      })),
      state('closed', style({
        height: '100px',
        opacity: 0.5,
        backgroundColor: 'red'
      })),
      transition('open => closed', animate('300ms ease-out')),
      transition('closed => open', animate('300ms ease-in'))
    ])
  ]
})
export class SimpleAnimationComponent {
  isOpen = true;
}
Animation triggers: The [@triggerName] syntax binds an animation trigger to a component property.

Animation Concepts

States

// Named states
state('active', style({ transform: 'scale(1.1)' }))
state('inactive', style({ transform: 'scale(1)' }))

// Wildcard state (matches any state)
state('*', style({ opacity: 1 }))

// Void state (element entering/leaving DOM)
state('void', style({ opacity: 0 }))

Transitions

// Specific transitions
transition('inactive => active', animate('200ms'))

// Any state change
transition('* => *', animate('500ms'))

// Entering/leaving DOM
transition(':enter', animate('300ms'))  // void => *
transition(':leave', animate('300ms'))  // * => void

// Multiple transitions
transition(':enter, :leave', animate('300ms'))

Styles

// Static styles
style({ backgroundColor: 'red', transform: 'translateX(0)' })

// Styles with parameters
style({ transform: 'translateX({ distance }})', backgroundColor: '{{ color }}' })

// Using function (for dynamic values)
style(({ toState }) => ({
  backgroundColor: toState === 'error' ? 'red' : 'green'
}))
Performance: Angular only animates properties that are animatable (transform, opacity, etc.). Avoid animating layout properties like width/height.

Complex Animations

animations: [
  trigger('slideInOut', [
    // Slide in from right
    transition(':enter', [
      style({ transform: 'translateX(100%)', opacity: 0 }),
      animate('300ms ease-out', style({ transform: 'translateX(0)', opacity: 1 }))
    ]),
    
    // Slide out to left
    transition(':leave', [
      animate('300ms ease-in', style({ transform: 'translateX(-100%)', opacity: 0 }))
    ])
  ]),
  
  trigger('fadeInOut', [
    transition(':enter', [
      style({ opacity: 0 }),
      animate('200ms', style({ opacity: 1 }))
    ]),
    transition(':leave', [
      animate('200ms', style({ opacity: 0 }))
    ])
  ])
]

Keyframe Animations

import { keyframes } from '@angular/animations';

animations: [
  trigger('bounce', [
    transition(':enter', [
      animate('1s', keyframes([
        style({ transform: 'translateY(-100%)', offset: 0 }),
        style({ transform: 'translateY(0)', offset: 0.5 }),
        style({ transform: 'translateY(-30px)', offset: 0.7 }),
        style({ transform: 'translateY(0)', offset: 1.0 })
      ]))
    ])
  ]),
  
  trigger('shake', [
    transition('* => error', [
      animate('0.5s', keyframes([
        style({ transform: 'translateX(0)', offset: 0 }),
        style({ transform: 'translateX(-10px)', offset: 0.2 }),
        style({ transform: 'translateX(10px)', offset: 0.4 }),
        style({ transform: 'translateX(-10px)', offset: 0.6 }),
        style({ transform: 'translateX(10px)', offset: 0.8 }),
        style({ transform: 'translateX(0)', offset: 1.0 })
      ]))
    ])
  ])
]

Route Animations

// app.component.ts
@Component({
  template: `
    <div [@routeAnimation]="getRouteAnimationData()">
      <router-outlet></router-outlet>
    </div>
  `,
  animations: [
    trigger('routeAnimation', [
      transition('* <=> *', [
        style({ position: 'relative' }),
        query(':enter, :leave', [
          style({ position: 'absolute', width: '100%' })
        ], { optional: true }),
        query(':enter', [
          style({ opacity: 0, transform: 'translateX(100%)' })
        ], { optional: true }),
        group([
          query(':leave', [
            animate('300ms ease-out', style({ opacity: 0, transform: 'translateX(-100%)' }))
          ], { optional: true }),
          query(':enter', [
            animate('300ms ease-out', style({ opacity: 1, transform: 'translateX(0)' }))
          ], { optional: true })
        ])
      ])
    ])
  ]
})
export class AppComponent {
  getRouteAnimationData() {
    return this.activatedRoute.snapshot.firstChild?.routeConfig?.path;
  }
}

Animation Callbacks

@Component({
  template: `
    <div 
      [@fadeAnimation]="animationState"
      (@fadeAnimation.start)="onAnimationStart($event)"
      (@fadeAnimation.done)="onAnimationDone($event)"
    >
      Animated content
    </div>
  `
})
export class AnimationCallbackComponent {
  animationState = 'visible';
  
  onAnimationStart(event: AnimationEvent) {
    console.log('Animation started:', event);
  }
  
  onAnimationDone(event: AnimationEvent) {
    console.log('Animation completed:', event);
    // Access final state
    console.log('To state:', event.toState);
  }
}

Reusable Animations

// animations/fade.animations.ts
import { trigger, transition, style, animate } from '@angular/animations';

export const fadeInOut = trigger('fadeInOut', [
  transition(':enter', [
    style({ opacity: 0 }),
    animate('200ms', style({ opacity: 1 }))
  ]),
  transition(':leave', [
    animate('200ms', style({ opacity: 0 }))
  ])
]);

export const slideInRight = trigger('slideInRight', [
  transition(':enter', [
    style({ transform: 'translateX(100%)', opacity: 0 }),
    animate('300ms ease-out', style({ transform: 'translateX(0)', opacity: 1 }))
  ]),
  transition(':leave', [
    animate('300ms ease-in', style({ transform: 'translateX(100%)', opacity: 0 }))
  ])
]);

// Use in component
import { fadeInOut, slideInRight } from '../animations/fade.animations';

@Component({
  animations: [fadeInOut, slideInRight],
  template: `
    <div [@fadeInOut]="show" [@slideInRight]="show">
      Reusable animation
    </div>
  `
})

Best Practices

  • Keep animations short - 200-500ms is ideal for UI feedback
  • Use hardware acceleration - Animate transform and opacity
  • Respect user preferences - Check for reduced motion preference
  • Test performance - Avoid animations on many elements
  • Use state machines - Define clear states and transitions
// Respect reduced motion preference
@media (prefers-reduced-motion: reduce) {
  * {
    animation: none !important;
    transition: none !important;
  }
}

// Or in Angular
animations: [
  trigger('fadeIn', [
    transition(':enter', [
      style({ opacity: 0 }),
      animate('{{ duration }}ms', style({ opacity: 1 }))
    ], { params: { duration: 200 } })
  ])
]
