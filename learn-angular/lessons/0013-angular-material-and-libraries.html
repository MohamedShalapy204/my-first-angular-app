Angular Material & Component Libraries

You can build complete applications. Now let's learn how to use Angular Material and other component libraries to create professional-looking UIs quickly.

Key Insight: Angular Material provides accessible, customizable UI components following Material Design. It's not just about looks—it's about building accessible, maintainable interfaces.

Using Components

// Import specific components
import { MatButtonModule } from '@angular/material/button';
import { MatCardModule } from '@angular/material/card';
import { MatInputModule } from '@angular/material/input';
import { MatIconModule } from '@angular/material/icon';

@Component({
  imports: [
    MatButtonModule,
    MatCardModule,
    MatInputModule,
    MatIconModule
  ],
  template: `
    <mat-card>
      <mat-card-header>
        <mat-card-title>User Profile</mat-card-title>
      </mat-card-header>
      
      <mat-card-content>
        <mat-form-field>
          <mat-label>Name</mat-label>
          <input matInput [formControl]="nameControl">
        </mat-form-field>
      </mat-card-content>
      
      <mat-card-actions>
        <button mat-raised-button color="primary" (click)="save()">
          Save
        </button>
        <button mat-button>Cancel</button>
      </mat-card-actions>
    </mat-card>
  `
})
export class UserProfileComponent {
  nameControl = new FormControl('');
}

Theming

// styles.scss
@use '@angular/material' as mat;

$primary: mat.m2-define-palette(mat.$m2-indigo-palette);
$accent: mat.m2-define-palette(mat.$m2-pink-palette, A200, A100, A400);
$warn: mat.m2-define-palette(mat.$m2-red-palette);

$theme: mat.m2-define-light-theme((
  color: (
    primary: $primary,
    accent: $accent,
    warn: $warn,
  ),
  typography: mat.m2-define-typography-config(),
));

@include mat.core();
@include mat.all-component-themes($theme);

// Custom theme
$custom-theme: mat.m2-define-light-theme((
  color: (
    primary: mat.m2-define-palette(mat.$m2-blue-grey-palette),
    accent: mat.m2-define-palette(mat.$m2-amber-palette),
  )
));

// Apply to specific component
.custom-card {
  @include mat.card-theme($custom-theme);
}
Theming: Angular Material uses SASS for theming. Create custom palettes or use built-in ones. Apply themes globally or per-component.

Dialogs

// Dialog component
@Component({
  template: `
    <h2 mat-dialog-title>{{ data.title }}</h2>
    <mat-dialog-content>
      <p>{{ data.message }}</p>
    </mat-dialog-content>
    <mat-dialog-actions>
      <button mat-button [mat-dialog-close]="false">Cancel</button>
      <button mat-raised-button color="primary" [mat-dialog-close]="true">
        Confirm
      </button>
    </mat-dialog-actions>
  `
})
export class ConfirmDialogComponent {
  static inject = [MAT_DIALOG_DATA];
  
  constructor(@Inject(MAT_DIALOG_DATA) public data: any) {}
}

// Opening dialog
@Component({...})
export class AppComponent {
  private dialog = inject(MatDialog);
  
  openConfirmDialog() {
    const dialogRef = this.dialog.open(ConfirmDialogComponent, {
      width: '400px',
      data: { title: 'Confirm Action', message: 'Are you sure?' }
    });
    
    dialogRef.afterClosed().subscribe(result => {
      if (result) {
        // User confirmed
        this.performAction();
      }
    });
  }
}

Snackbar Notifications

@Component({...})
export class AppComponent {
  private snackBar = inject(MatSnackBar);
  
  showSuccess(message: string) {
    this.snackBar.open(message, 'Close', {
      duration: 3000,
      horizontalPosition: 'end',
      verticalPosition: 'top',
      panelClass: ['success-snackbar']
    });
  }
  
  showError(message: string) {
    this.snackBar.open(message, 'Retry', {
      duration: 5000,
      panelClass: ['error-snackbar']
    }).onAction().subscribe(() => {
      this.retryOperation();
    });
  }
}

CDK (Component Dev Kit)

// Angular CDK provides low-level utilities
import { CdkDragDrop, moveItemInArray } from '@angular/cdk/drag-drop';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';

@Component({
  template: `
    <div cdkDropList (cdkDropListDropped)="drop($event)">
      @for (item of items; track item.id) {
        <div cdkDrag>{{ item.name }}</div>
      }
    </div>
  `
})
export class DragDropComponent {
  items = [
    { id: 1, name: 'Item 1' },
    { id: 2, name: 'Item 2' },
    { id: 3, name: 'Item 3' }
  ];
  
  drop(event: CdkDragDrop<Item[]>) {
    moveItemInArray(this.items, event.previousIndex, event.currentIndex);
  }
}

// Responsive layout
@Component({...})
export class ResponsiveComponent {
  isMobile = signal(false);
  
  constructor() {
    inject(BreakpointObserver)
      .observe([Breakpoints.Handset])
      .subscribe(result => {
        this.isMobile.set(result.matches);
      });
  }
}

Best Practices

  • Import only what you need - Tree-shake unused components
  • Use CDK for custom components - Don't reinvent drag-drop, modals, etc.
  • Customize themes - Don't use default Material look
  • Accessibility first - Material components are accessible by default
  • Consider bundle size - Some libraries are heavy
