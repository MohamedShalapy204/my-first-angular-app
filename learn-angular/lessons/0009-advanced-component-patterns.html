Advanced Component Patterns

You can build and test components. Now let's explore advanced patterns that make your components more flexible, reusable, and powerful.

Key Insight: Advanced patterns like content projection, dynamic components, and host elements let you build truly reusable UI libraries and complex component architectures.

Content Projection

Let parent components project content into your component:

// Card component that projects content
@Component({
  selector: 'app-card',
  template: `
    <div class="card">
      @if (title) {
        <div class="card-header">
          <h3>{{ title }}</h3>
        </div>
      }
      
      <div class="card-body">
        <ng-content></ng-content>
      </div>
      
      @if (footer) {
        <div class="card-footer">
          <ng-content select="[card-footer]"></ng-content>
        </div>
      }
    </div>
  `
})
export class CardComponent {
  title = input<string>();
  footer = input(false);
}

// Usage
<app-card [title]="'User Info'" [footer]="true">
  <p>This is the main content</p>
  
  <div card-footer>
    <button>Save</button>
    <button>Cancel</button>
  </div>
</app-card>

Multiple Content Slots

<!-- Template with multiple projection points -->
<ng-template #header>
  <ng-content select="[header]"></ng-content>
</ng-template>

<ng-template #footer>
  <ng-content select="[footer]"></ng-content>
</ng-template>

<!-- Default content when nothing projected -->
<ng-content select="[header]">Default Header</ng-content>
select attribute: Use CSS selectors to target specific projected content. ng-content select=".my-class" only projects elements with that class.

Dynamic Components

Load components dynamically at runtime:

// Create a component registry
const componentRegistry: Record<string, Type<any>> = {
  'user-card': UserCardComponent,
  'product-card': ProductCardComponent,
  'alert-box': AlertBoxComponent
};

@Component({
  template: `
    <div #container></div>
  `
})
export class DynamicHostComponent implements AfterViewInit {
  @ViewChild('container', { read: ViewContainerRef }) 
  container!: ViewContainerRef;
  
  loadComponent(componentType: string) {
    this.container.clear();
    
    const component = componentRegistry[componentType];
    if (component) {
      const ref = this.container.createComponent(component);
      
      // Set inputs
      ref.setInput('data', { id: 1, name: 'Example' });
      
      // Listen to outputs
      ref.instance.action.subscribe((event: any) => {
        console.log('Component action:', event);
      });
    }
  }
}

Using ngComponentOutlet (Simpler)

@Component({
  template: `
    <ng-container 
      *ngComponentOutlet="currentComponent"
      [ngComponentOutletInputs]="componentInputs"
      (ngComponentOutletOutput)="handleOutput($event)"
    ></ng-container>
  `
})
export class DynamicHostComponent {
  currentComponent = signal<Type<any>>(UserCardComponent);
  componentInputs = signal({ data: { id: 1 } });
  
  handleOutput(event: any) {
    console.log('Output:', event);
  }
}
Lazy Loading: Dynamic components work well with lazy loading. Only load components when needed.

Host Elements

Access the host element (where component is placed):

@Component({
  selector: '[appHighlight]',
  host: {
    'class': 'highlighted',
    '[class.active]': 'isActive',
    '(click)': 'onClick()',
    'role': 'button',
    'tabindex': '0'
  },
  template: `...`
})
export class HighlightDirective {
  isActive = signal(false);
  
  @HostBinding('class.highlighted')
  isHighlighted = true;
  
  @HostListener('mouseenter')
  onMouseEnter() {
    this.isActive.set(true);
  }
  
  @HostListener('mouseleave')
  onMouseLeave() {
    this.isActive.set(false);
  }
}

Component Inheritance

// Base component
@Component({
  template: `
    <div class="base-layout">
      <header>
        <ng-content select="[header]"></ng-content>
      </header>
      
      <main>
        <ng-content></ng-content>
      </main>
      
      <footer>
        <ng-content select="[footer]"></ng-content>
      </footer>
    </div>
  `
})
export class BaseLayoutComponent {
  title = input<string>();
}

// Extended component
@Component({
  selector: 'app-admin-layout',
  template: `
    <div class="admin-wrapper">
      <app-sidebar />
      <div class="main-content">
        <ng-content />
      </div>
    </div>
  `
})
export class AdminLayoutComponent extends BaseLayoutComponent {
  // Inherits title input
  sidebarCollapsed = signal(false);
}

Component Communication Patterns

1. Bidirectional Binding with Signals

// Child component
@Component({...})
export class SearchInputComponent {
  value = model<string>(''); // Two-way bindable signal
  
  onInput(event: Event) {
    this.value.set((event.target as HTMLInputElement).value);
  }
}

// Parent component
@Component({
  template: `
    <app-search-input [(value)]="searchQuery" />
    <p>Searching for: {{ searchQuery() }}</p>
  `
})
export class ParentComponent {
  searchQuery = signal('');
}

2. Component State Machine

type ComponentState = 'idle' | 'loading' | 'success' | 'error';

@Component({...})
export class AsyncComponent {
  state = signal<ComponentState>('idle');
  data = signal<any>(null);
  error = signal<string | null>(null);
  
  async loadData() {
    this.state.set('loading');
    
    try {
      this.data.set(await this.api.getData());
      this.state.set('success');
    } catch (err) {
      this.error.set(err.message);
      this.state.set('error');
    }
  }
  
  retry() {
    this.state.set('idle');
    this.loadData();
  }
}

Lazy Component Loading

// Lazy load heavy components
@Component({
  template: `
    @if (showChart()) {
      <app-heavy-chart [data]="chartData()" />
    } @else {
      <button (click)="showChart.set(true)">
        Load Chart
      </button>
    }
  `
})
export class DashboardComponent {
  showChart = signal(false);
  chartData = signal([]);
}

// Or with dynamic imports
const routes: Routes = [
  {
    path: 'dashboard',
    loadComponent: () => import('./dashboard.component')
      .then(m => m.DashboardComponent)
  }
];

Best Practices

  • Keep components focused - One responsibility per component
  • Use content projection for composition - Let parents control content
  • Prefer signals over decorators - More reactive, less boilerplate
  • Use model() for two-way binding - Cleaner than @Output for simple cases
  • Consider lazy loading - Load components only when needed
