Performance Optimization

You can build feature-rich applications. Now let's learn how to make them fast and efficient for users.

Key Insight: Performance optimization isn't about premature optimization—it's about understanding Angular's change detection, rendering, and bundle optimization to create smooth user experiences.

Change Detection Strategies

Default vs OnPush

// Default change detection (checks entire component tree)
@Component({
  selector: 'app-default',
  template: `...`
})
export class DefaultComponent {}

// OnPush (only checks when inputs change or manual trigger)
@Component({
  selector: 'app-on-push',
  changeDetection: ChangeDetectionStrategy.OnPush,
  template: `...`
})
export class OnPushComponent {
  // Only updates when:
  // 1. @Input references change
  // 2. Signal values change
  // 3. Explicit markForCheck() called
  // 4. Async pipe receives new value
}
OnPush gotcha: If you mutate input objects instead of creating new references, OnPush won't detect changes. Always use immutable patterns or signals.

Manual Change Detection

import { ChangeDetectorRef } from '@angular/core';

@Component({...})
export class ManualDetectionComponent {
  private cdr = inject(ChangeDetectorRef);
  
  externalUpdate() {
    // When external events update state
    this.data = fetchData();
    this.cdr.markForCheck(); // Tell Angular to check
    
    // Or detach and reattach for performance
    this.cdr.detach();
    this.updateData();
    this.cdr.reattach();
  }
}

Signals for Performance

@Component({...})
export class OptimizedComponent {
  // Signals automatically track dependencies
  items = signal<Item[]>([]);
  filter = signal('');
  
  // Computed only recalculates when dependencies change
  filteredItems = computed(() => 
    this.items().filter(item => 
      item.name.includes(this.filter())
    )
  );
  
  // Effect runs only when its dependencies change
  constructor() {
    effect(() => {
      // Only re-runs when items() or filter() changes
      console.log('Items count:', this.filteredItems().length);
    });
  }
}

Lazy Loading

Route-level Lazy Loading

const routes: Routes = [
  // Lazy load entire feature
  {
    path: 'admin',
    loadChildren: () => import('./admin/admin.routes')
      .then(m => m.ADMIN_ROUTES)
  },
  
  // Lazy load single component
  {
    path: 'dashboard',
    loadComponent: () => import('./dashboard.component')
      .then(m => m.DashboardComponent)
  }
];

Preloading Strategies

// Preload all lazy modules (good for small apps)
provideRouter(routes, withPreloading(PreloadAllModules))

// Preload based on network condition
provideRouter(routes, withPreloading(PreloadingStrategy))

// Custom preloading strategy
export class SelectivePreloadingStrategy implements PreloadingStrategy {
  preload(route: Route, load: () => Observable<any>): Observable<any> {
    if (route.data?.['preload']) {
      return load();
    }
    return of(null);
  }
}
Bundle analysis: Use ng build --stats-json and webpack-bundle-analyzer to identify large dependencies.

OnPush with Async Pipe

@Component({
  changeDetection: ChangeDetectionStrategy.OnPush,
  template: `
    @if (users$ | async; as users) {
      @for (user of users; track user.id) {
        <div>{{ user.name }}</div>
      }
    }
  `
})
export class AsyncComponent {
  // Async pipe handles subscription and change detection
  users$ = this.userService.getUsers().pipe(
    tap(() => this.cdr.markForCheck()) // Ensure OnPush updates
  );
  
  constructor(
    private userService: UserService,
    private cdr: ChangeDetectorRef
  ) {}
}

Virtual Scrolling

import { ScrollingModule } from '@angular/cdk/scrolling';

@Component({
  imports: [ScrollingModule],
  template: `
    <!-- Only renders visible items -->
    <cdk-virtual-scroll-viewport itemSize="50" class="viewport">
      <div *cdkVirtualFor="let item of items" class="item">
        {{ item.name }}
      </div>
    </cdk-virtual-scroll-viewport>
  `
})
export class VirtualListComponent {
  items = Array.from({ length: 10000 }, (_, i) => ({
    id: i,
    name: `Item ${i}`
  }));
}

Memory Leaks

// ❌ Memory leak - subscription never cleaned up
ngOnInit() {
  this.interval$ = interval(1000).subscribe(x => console.log(x));
}

// ✅ Proper cleanup
ngOnInit() {
  this.interval$ = interval(1000).pipe(
    takeUntilDestroyed() // Auto-cleanup on destroy
  ).subscribe(x => console.log(x));
}

// Or manual cleanup
private destroy$ = new Subject<void>();

ngOnInit() {
  interval(1000)
    .pipe(takeUntil(this.destroy$))
    .subscribe(x => console.log(x));
}

ngOnDestroy() {
  this.destroy$.next();
  this.destroy$.complete();
}
