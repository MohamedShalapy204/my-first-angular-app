Internationalization (i18n)

You can build and deploy applications. Now let's learn how to make them accessible to users worldwide with Angular's internationalization features.

Key Insight: i18n isn't just about translating text—it's about adapting your app for different languages, currencies, date formats, and cultural conventions.

Angular i18n Setup

# Add i18n support
ng add @angular/localize

# Extract translatable text
ng extract-i18n

Marking Text for Translation

<!-- In templates -->
<h1 i18n="@@welcomeHeader">Welcome to our app</h1>
<p i18n="@@introText">This is a sample application</p>

<!-- With description for translators -->
<p i18n="@@userCount|Number of users in the system">
  {{ userCount }} users
</p>

<!-- Attribute translation -->
<img [alt]="'Logo'" i18n-alt="@@logoAlt">

<!-- Placeholder translation -->
<input placeholder="Enter email" i18n-placeholder="@@emailPlaceholder">

Translation Files

// messages.xlf (generated by ng extract-i18n)
<xliff version="1.2">
  <file source-language="en" target-language="fr">
    <body>
      <trans-unit id="welcomeHeader">
        <source>Welcome to our app</source>
        <target>Bienvenue dans notre application</target>
      </trans-unit>
      <trans-unit id="introText">
        <source>This is a sample application</source>
        <target>Ceci est une application d'exemple</target>

</xliff>

Translation Service

import { TranslateService } from '@ngx-translate/core';

@Component({
  template: `
    <h1>{{ 'WELCOME' | translate }}</h1>
    <p>{{ 'USER_COUNT' | translate: { count: userCount } }}</p>
    
    <select (change)="switchLanguage($event.target.value)">
      @for (lang of languages; track lang.code) {
        <option [value]="lang.code">{{ lang.name }}</option>
      }
    </select>
  `
})
export class AppComponent {
  private translate = inject(TranslateService);
  
  languages = [
    { code: 'en', name: 'English' },
    { code: 'fr', name: 'Français' },
    { code: 'es', name: 'Español' },
    { code: 'de', name: 'Deutsch' }
  ];
  
  userCount = 42;
  
  constructor() {
    // Set default language
    this.translate.setDefaultLang('en');
    this.translate.use('en');
  }
  
  switchLanguage(lang: string) {
    this.translate.use(lang);
  }
}

Date and Number Formatting

import { DatePipe, CurrencyPipe, DecimalPipe } from '@angular/common';

@Component({
  template: `
    <!-- Date formatting -->
    <p>{{ today | date:'fullDate' }}</p>
    <p>{{ today | date:'short' }}</p>
    
    <!-- Currency formatting -->
    <p>{{ price | currency:'USD':'symbol':'1.2-2' }}</p>
    <p>{{ price | currency:'EUR':'symbol':'1.2-2' }}</p>
    
    <!-- Number formatting -->
    <p>{{ number | number:'1.0-0' }}</p>
  `
})
export class FormattingComponent {
  today = new Date();
  price = 1234.56;
  number = 1234567.89;
}
Locale: Angular pipes automatically format based on the current locale. Set it with LOCALE_ID.

RTL Support

// Detect and apply RTL
@Component({
  template: `
    <div [dir]="isRtl ? 'rtl' : 'ltr'">
      <p>This text will be right-aligned for RTL languages</p>
    </div>
  `,
  styles: [`
    :host {
      display: block;
    }
    
    [dir="rtl"] {
      text-align: right;
    }
    
    [dir="rtl"] .sidebar {
      margin-right: 0;
      margin-left: auto;
    }
  `]
})
export class RtlComponent {
  private translate = inject(TranslateService);
  
  isRtl = computed(() => {
    const lang = this.translate.currentLang;
    return ['ar', 'he', 'fa'].includes(lang);
  });
}

Pluralization

<!-- i18n plural handling -->
<p i18n="@@itemCount">
  {itemCount, plural, 
    =0 {No items} 
    one {One item} 
    other {{itemCount} items}
  }
</p>

<!-- With ngx-translate -->
<p>{{ 'ITEM_COUNT' | translate:{ count: itemCount } }}</p>

// messages.fr.xlf
<trans-unit id="itemCount">
  <source>{itemCount, plural, =0 {No items} one {One item} other {{itemCount} items}}</source>
  <target>{itemCount, plural, =0 {Aucun élément} one {Un élément} other {{itemCount} éléments}}</target>
</trans-unit>

Translation Extraction Workflow

# 1. Extract source text
ng extract-i18n --output-path src/locale

# 2. Send messages.xlf to translators

# 3. Create translation files
# src/locale/messages.fr.xlf
# src/locale/messages.es.xlf

# 4. Build for specific locale
ng build --localize=fr
ng build --localize=es

# 5. Deploy separate builds per locale

Lazy-Loaded Translations

// Load translations on demand
@Injectable({ providedIn: 'root' })
export class TranslationLoader implements TranslateLoader {
  private http = inject(HttpClient);
  
  getTranslation(lang: string): Observable<any> {
    return this.http.get(`/assets/i18n/${lang}.json`).pipe(
      tap(translations => {
        console.log(`Loaded ${Object.keys(translations).length} translations for ${lang}`);
      })
    );
  }
}

// Configure with TranslateModule.forRoot
TranslateModule.forRoot({
  loader: {
    provide: TranslateLoader,
    useClass: TranslationLoader
  }
})
