Routing & Navigation

You can build components and share services. Now let's learn how to connect them into a navigable application with Angular's router.

Key Insight: Angular Router turns your single-page app into a multi-view application. It maps URLs to components, enabling navigation without full page reloads.

Basic Router Setup

// app.routes.ts
import { Routes } from '@angular/router';
import { HomeComponent } from './home/home.component';
import { AboutComponent } from './about/about.component';
import { UserComponent } from './user/user.component';

export const routes: Routes = [
  { path: '', component: HomeComponent },
  { path: 'about', component: AboutComponent },
  { path: 'user/:id', component: UserComponent },
  { path: '**', redirectTo: '' }  // Wildcard route
];
// app.config.ts
import { ApplicationConfig, provideRouter } from '@angular/router';
import { routes } from './app.routes';

export const appConfig: ApplicationConfig = {
  providers: [
    provideRouter(routes),
    // ... other providers
  ]
};
<!-- app.component.html -->
<nav>
  <a routerLink="/">Home</a>
  <a routerLink="/about">About</a>
  <a [routerLink]="['/user', userId]">User</a>
</nav>

<!-- Where routed components render -->
<router-outlet></router-outlet>
routerLink vs href: Use routerLink for Angular routes (no page reload). Use href for external links.

Route Parameters

// Route with parameter
{ path: 'user/:id', component: UserComponent }

// Access in component
@Component({...})
export class UserComponent implements OnInit {
  private route = inject(ActivatedRoute);
  userId = signal<string>('');
  
  ngOnInit() {
    // Using signals (modern approach)
    this.route.params.subscribe(params => {
      this.userId.set(params['id']);
    });
    
    // Or using snapshot (for static routes)
    const id = this.route.snapshot.params['id'];
  }
}

Query Parameters

// URL: /search?q=angular&page=2
@Component({...})
export class SearchComponent implements OnInit {
  private route = inject(ActivatedRoute);
  searchQuery = signal('');
  page = signal(1);
  
  ngOnInit() {
    this.route.queryParams.subscribe(params => {
      this.searchQuery.set(params['q'] || '');
      this.page.set(params['page'] || 1);
    });
  }
  
  updateSearch(query: string) {
    this.router.navigate(['/search'], {
      queryParams: { q: query, page: 1 }
    });
  }
}
Snapshot vs Subscribe: Use snapshot for static routes, subscribe for dynamic routes that might change without component reload.

Navigation Programmatic

@Component({...})
export class LoginComponent {
  private router = inject(Router);
  
  async login(credentials: LoginRequest) {
    try {
      await this.authService.login(credentials);
      // Navigate after successful login
      this.router.navigate(['/dashboard']);
      
      // With query params
      this.router.navigate(['/dashboard'], {
        queryParams: { welcome: true }
      });
      
      // Relative navigation
      this.router.navigate(['../sibling'], { relativeTo: this.route });
    } catch (error) {
      console.error('Login failed');
    }
  }
}

Route Guards

Guards control access to routes:

// CanActivate guard
import { CanActivateFn } from '@angular/router';

export const authGuard: CanActivateFn = (route, state) => {
  const authService = inject(AuthService);
  const router = inject(Router);
  
  if (authService.isAuthenticated()) {
    return true;
  }
  
  // Redirect to login
  return router.createUrlTree(['/login'], {
    queryParams: { returnUrl: state.url }
  });
};

// Usage in routes
{ path: 'dashboard', component: DashboardComponent, canActivate: [authGuard] }

Common Guard Types

Guard Purpose When It Runs
canActivate Control access to route Before route activation
canDeactivate Prevent leaving route Before route deactivation
canActivateChild Control child route access Before child route activation
resolve Pre-fetch data Before route activation

Lazy Loading

Load components only when needed:

// Lazy load entire feature module
const routes: Routes = [
  {
    path: 'admin',
    loadChildren: () => import('./admin/admin.routes')
      .then(m => m.ADMIN_ROUTES)
  },
  {
    path: 'shop',
    loadComponent: () => import('./shop/shop.component')
      .then(m => m.ShopComponent)
  }
];
Performance: Lazy loading splits your app into smaller bundles. Users only download code for the routes they visit.

Nested Routes

// Parent route with children
const routes: Routes = [
  {
    path: 'user/:id',
    component: UserComponent,
    children: [
      { path: 'profile', component: UserProfileComponent },
      { path: 'settings', component: UserSettingsComponent },
      { path: '', redirectTo: 'profile', pathMatch: 'full' }
    ]
  }
];

// Parent template with child outlet
@Component({
  template: `
    <div class="user-layout">
      <nav>
        <a [routerLink]="['profile']">Profile</a>
        <a [routerLink]="['settings']">Settings</a>
      </nav>
      
      <!-- Child routes render here -->
      <router-outlet></router-outlet>
    </div>
  `
})
export class UserComponent {}

Route Data & Resolve

// Static route data
{ 
  path: 'about', 
  component: AboutComponent,
  data: { title: 'About Us', showHeader: true }
}

// Access in component
ngOnInit() {
  this.route.data.subscribe(data => {
    this.title = data['title'];
  });
}

// Resolve: pre-fetch data before route loads
export const userResolver: ResolveFn<User> = (route) => {
  const userService = inject(UserService);
  const userId = route.params['id'];
  return userService.getUser(userId);
};

// Route with resolver
{
  path: 'user/:id',
  component: UserComponent,
  resolve: { user: userResolver }
}

// Component receives resolved data
@Component({...})
export class UserComponent implements OnInit {
  user = signal<User>(null);
  
  ngOnInit() {
    const resolvedUser = this.route.snapshot.data['user'];
    this.user.set(resolvedUser);
  }
}

Common Patterns

1. Not Found Page

// Wildcard route (must be last)
{ path: '**', component: NotFoundComponent }

// Or redirect to home
{ path: '**', redirectTo: '' }

2. Redirect Routes

// Redirect /old-path to /new-path
{ path: 'old-path', redirectTo: 'new-path', pathMatch: 'full' }

// Default route
{ path: '', redirectTo: '/home', pathMatch: 'full' }

3. Route Animation

// In component
@Component({
  animations: [
    trigger('routeAnimation', [
      transition('* <=> *', [
        style({ position: 'relative' }),
        query(':enter', [
          style({ position: 'absolute', width: '100%' })
        ], { optional: true }),
        group([
          query(':leave', [
            animate('300ms ease-out', style({ opacity: 0 }))
          ], { optional: true }),
          query(':enter', [
            style({ opacity: 0 }),
            animate('300ms ease-out', style({ opacity: 1 }))
          ], { optional: true })
        ])
      ])
    ])
  ]
})
